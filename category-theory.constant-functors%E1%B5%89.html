Idea

A constant functor is a functor F : C → D that is constant at an object d ∈ D and the identity morphism at that object.

Definition

Constant functors between precategories

module _
  {l1ᵉ l2ᵉ l3ᵉ l4ᵉ : Level} (Cᵉ : Precategoryᵉ l1ᵉ l2ᵉ) (Dᵉ : Precategoryᵉ l3ᵉ l4ᵉ)
  (dᵉ : obj-Precategoryᵉ Dᵉ)
  where

  constant-functor-Precategoryᵉ : functor-Precategoryᵉ Cᵉ Dᵉ
  pr1ᵉ constant-functor-Precategoryᵉ _ = dᵉ
  pr1ᵉ (pr2ᵉ constant-functor-Precategoryᵉ) _ = id-hom-Precategoryᵉ Dᵉ
  pr1ᵉ (pr2ᵉ (pr2ᵉ constant-functor-Precategoryᵉ)) _ _ =
    invᵉ (left-unit-law-comp-hom-Precategoryᵉ Dᵉ (id-hom-Precategoryᵉ Dᵉ))
  pr2ᵉ (pr2ᵉ (pr2ᵉ constant-functor-Precategoryᵉ)) = refl-htpyᵉ

Constant functors between large precategories

module _
  {αCᵉ αDᵉ : Level → Level} {βCᵉ βDᵉ : Level → Level → Level}
  (Cᵉ : Large-Precategoryᵉ αCᵉ βCᵉ) (Dᵉ : Large-Precategoryᵉ αDᵉ βDᵉ)
  {lᵉ : Level} (dᵉ : obj-Large-Precategoryᵉ Dᵉ lᵉ)
  where

  constant-functor-Large-Precategoryᵉ : functor-Large-Precategoryᵉ (λ _ → lᵉ) Cᵉ Dᵉ
  obj-functor-Large-Precategoryᵉ constant-functor-Large-Precategoryᵉ _ = dᵉ
  hom-functor-Large-Precategoryᵉ constant-functor-Large-Precategoryᵉ _ =
    id-hom-Large-Precategoryᵉ Dᵉ
  preserves-comp-functor-Large-Precategoryᵉ constant-functor-Large-Precategoryᵉ
    _ _ =
    invᵉ
      ( left-unit-law-comp-hom-Large-Precategoryᵉ Dᵉ (id-hom-Large-Precategoryᵉ Dᵉ))
  preserves-id-functor-Large-Precategoryᵉ constant-functor-Large-Precategoryᵉ =
    reflᵉ

Properties

The constant functor induces a functor into the precategory of functors

module _
  {l1ᵉ l2ᵉ l3ᵉ l4ᵉ : Level} (Cᵉ : Precategoryᵉ l1ᵉ l2ᵉ) (Dᵉ : Precategoryᵉ l3ᵉ l4ᵉ)
  where

  natural-transformation-constant-functor-Precategoryᵉ :
    {xᵉ yᵉ : obj-Precategoryᵉ Dᵉ}
    (fᵉ : hom-Precategoryᵉ Dᵉ xᵉ yᵉ) →
    natural-transformation-Precategoryᵉ Cᵉ Dᵉ
      (constant-functor-Precategoryᵉ Cᵉ Dᵉ xᵉ)
      (constant-functor-Precategoryᵉ Cᵉ Dᵉ yᵉ)
  pr1ᵉ (natural-transformation-constant-functor-Precategoryᵉ fᵉ) _ = fᵉ
  pr2ᵉ (natural-transformation-constant-functor-Precategoryᵉ fᵉ) hᵉ =
    left-unit-law-comp-hom-Precategoryᵉ Dᵉ _ ∙ᵉ
    invᵉ (right-unit-law-comp-hom-Precategoryᵉ Dᵉ _)

  functor-constant-functor-Precategoryᵉ :
    functor-Precategoryᵉ Dᵉ (functor-precategory-Precategoryᵉ Cᵉ Dᵉ)
  pr1ᵉ functor-constant-functor-Precategoryᵉ xᵉ =
    constant-functor-Precategoryᵉ Cᵉ Dᵉ xᵉ
  pr1ᵉ (pr2ᵉ functor-constant-functor-Precategoryᵉ) fᵉ =
    natural-transformation-constant-functor-Precategoryᵉ fᵉ
  pr1ᵉ (pr2ᵉ (pr2ᵉ functor-constant-functor-Precategoryᵉ)) gᵉ fᵉ =
    eq-pair-Σᵉ
      ( reflᵉ)
      ( eq-is-propᵉ
        ( is-prop-is-natural-transformation-Precategoryᵉ Cᵉ Dᵉ
          ( constant-functor-Precategoryᵉ Cᵉ Dᵉ _)
          ( constant-functor-Precategoryᵉ Cᵉ Dᵉ _)
          ( _)))
  pr2ᵉ (pr2ᵉ (pr2ᵉ functor-constant-functor-Precategoryᵉ)) xᵉ =
    eq-pair-Σᵉ
      ( reflᵉ)
      ( eq-is-propᵉ
        ( is-prop-is-natural-transformation-Precategoryᵉ Cᵉ Dᵉ
          ( constant-functor-Precategoryᵉ Cᵉ Dᵉ _)
          ( constant-functor-Precategoryᵉ Cᵉ Dᵉ _)
          ( _)))
