Discrete precategories

Any set induces a discrete category whose objects are elements of the set and which contains no-nonidentity morphisms.

module _
  {lᵉ : Level} (Xᵉ : Setᵉ lᵉ)
  where

  discrete-precategory-Setᵉ : Precategoryᵉ lᵉ lᵉ
  discrete-precategory-Setᵉ =
    make-Precategoryᵉ
      ( type-Setᵉ Xᵉ)
      ( λ xᵉ yᵉ → set-Propᵉ (Id-Propᵉ Xᵉ xᵉ yᵉ))
      ( λ pᵉ qᵉ → qᵉ ∙ᵉ pᵉ)
      ( λ xᵉ → reflᵉ)
      ( λ hᵉ gᵉ fᵉ → invᵉ (assocᵉ fᵉ gᵉ hᵉ))
      ( λ _ → right-unitᵉ)
      ( λ _ → left-unitᵉ)
